# 启动方式
在服务器上安装如下应用：
```
yum install gcc -y
yum install zlib zlib-devel -y
yum install bzip2 bzip2-devel  -y
yum install ncurses ncurses-devel  -y
yum install readline readline-devel  -y
yum install openssl openssl-devel  -y
yum install xz lzma xz-devel  -y
yum install sqlite sqlite-devel  -y
yum install gdbm gdbm-devel  -y
yum install tk tk-devel  -y
yum install mysql-devel -y
yum install python-devel -y
yum install libffi-devel -y
yum install mariadb-server -y
yum install mariadb -y
yum install nginx -y
```

启动MariaDB服务和nginx服务
```
systemctl start MariaDB
systemctl enable mariadb
systemctl start nginx
systemctl enable nginx
```

在mysql中创建数据库，并在项目的fake_image_detector/settings.py中配置和数据库的连接。依次执行python manage.py makemigrations和python manage.py migrate迁移数据库，并用python manage.py superuser创建根管理员，执行python manage.py collectstatic收集静态资源。

生产环境建议配置uwsgi运行项目，参考的fake_image_detector_uwsgi.ini配置如下：
```
socket = 127.0.0.1:9000                                                                                                                  
chdir = /data/fake_image_detector                                                                                                        
wsgi-file = fake_image_detector/wsgi.py                                                                                                  
processes = 4                                                                                                                      
virtualenv = /envs/fake_image_detector/                                                                                                  
buffer-size = 32768                                                                                                                      
max-requests = 1000
```

配置nginx.conf文件
```
server {
    client_max_body_size 100M;

    listen       80 default_server; ###
    listen       [::]:80;
    server_name  _; ###
    # 前端 dist 所在目录
    root   /data/vue-project/;
    index  index.html;

    # Load configuration files for the default server block.
    # include /etc/nginx/default.d/*.conf;

    location /static {
        alias /data/allstatic/;
    }

    location /media {
        alias /data/media/;
    }

    # 所有 /api/ 前缀转给后端
    location ^~ /api/ {
        # allow  127.0.0.1;
        # deny   all;
        uwsgi_pass django;
        include uwsgi_params;
        uwsgi_read_timeout 1200s;
        uwsgi_send_timeout 1200s;
    }
    # WebSocket 配置: 处理 /ws/notifications/ 路由的 WebSocket 连接
    location /ws/notifications/ {
        proxy_pass http://127.0.0.1:9000;  # 替换为 Django WebSocket 服务的地址
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
    # 所有 /admin/ 前缀转给后端
    location ^~ /admin/ {
        # allow  127.0.0.1;
        # deny   all;
        uwsgi_pass django;
        include uwsgi_params;
    }
    ###
    location / {
        try_files $uri $uri/ /index.html;
    }
}


server {

运行如下脚本以启动服务。注意，要确保只有celery_ai与AI服务器连接，否则需重新启动。
```
#!/usr/bin/env bash                                                                                                                      
                                                                                                                                         
# —— 1. 杀掉旧的 uwsgi、redis、celery ——                                                                                                 
for name in "fake_image_detector_uwsgi.ini" "redis-server" "celery -A fake_image_detector.celery worker -Q cpu" "celery -A fake_image_det
ector.celery worker -Q ai" "celery -A fake_image_detector.celery worker"; do                                                             
  pids=$(pgrep -f "$name")                                                                                                               
  if [ -n "$pids" ]; then                                                                                                                
    echo "Killing existing process: $name → $pids"                                                                                       
    echo "$pids" | xargs kill -9                                                                                                         
    sleep 0.5                                                                                                                            
  fi                                                                                                                                     
done                                                                                                                                     
                                                                                                                                         
echo -e "\n\033[34m--- Starting uwsgi ---\033[0m"                                                                                        
#/envs/fake_image_detector/bin/uwsgi --ini fake_image_detector_uwsgi.ini & >/dev/null                                                    
nohup /envs/fake_image_detector/bin/uwsgi --ini fake_image_detector_uwsgi.ini > uwsgi.log 2>&1 &                                         
                                                                                                                                         
echo -e "\n\033[34m--- Starting redis-server ---\033[0m"                                                                                 
nohup redis-server > /dev/null 2>&1 &                                                                                                    
                                                                                                                                         
                                                                                                                                         
                                                                                                                                         
# —— 2. 修改 CONNECT 标志 ——                                                                                                             
FILE="core/call_figure_detection.py"                                                                                                     
                                                                                                                                         
# 检查文件中是否恰好只有一处 CONNECT = False                                                                                             
count=$(grep -c -E '^\s*CONNECT\s*=\s*False\s*$' "$FILE")                                                                                
if [ "$count" -ne 1 ]; then                                                                                                              
  echo "Error: expected exactly one 'CONNECT = False' in $FILE, but found $count."                                                       
  exit 1                                                                                                                                 
fi                                                                                                                                       
                                                                                                                                         
echo -e "\n\033[34m--- Enabling detection (CONNECT=True) ---\033[0m"                                                                     
sed -i 's/^\(\s*CONNECT\s*=\s*\)False\s*$/\1True/' "$FILE"                                                                               
                                                                                                                                         
# —— 3. 启动 celery worker ——                                                                                                            
echo -e "\n\033[34m--- Starting celery worker ---\033[0m"                                                                                
# Celery GPU worker                                                                                                                      
nohup celery -A fake_image_detector.celery worker -Q ai  --concurrency 1 --prefetch-multiplier 1 -n ai@%h --loglevel=info > celery_ai.log
 2>&1 &                                                                                                                                  
# Celery CPU worker                                                                                                                      
# nohup celery -A fake_image_detector.celery worker -Q cpu --concurrency 4 -n cpu@%h --loglevel=info > celery_cpu.log 2>&1 &             
# nohup celery -A fake_image_detector.celery worker --loglevel=info > celery.log 2>&1 &                                                  
                                                                                                                                         
sleep 10                                                                                                                                 
                                                                                                                                         
# —— 4. 恢复 CONNECT 标志 ——                                                                                                             
echo -e "\n\033[34m--- Restoring detection flag (CONNECT=False) ---\033[0m"                                                              
sed -i 's/^\(\s*CONNECT\s*=\s*\)True\s*$/\1False/' "$FILE"                                                                               
                                                                                                                                         
nohup celery -A fake_image_detector.celery worker -Q cpu --concurrency 4 -n cpu@%h --loglevel=info > celery_cpu.log 2>&1 &               
                                                                                                                                         
echo -e "\n\033[42;1mAll services started. \033[0m"
